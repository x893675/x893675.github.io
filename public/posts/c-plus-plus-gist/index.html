Thread::~Thread()
{
    Join();
    if (runnable_ != NULL)
    {
        delete runnable_;
    }
}
void Thread::Kill()
{
    if (alive_)
    {
        pthread_kill(thread_, 0);
    }
    alive_ = false;
}
void Thread::Join()
{
    if (alive_)
    {
        void *result = NULL;
        pthread_join(thread_, &result);
        pthread_detach(thread_);
        thread_ = 0;
    }
    alive_ = false;
}
void Thread::Run()
{
    runnable_->Run(this);
}
void *Thread::ThreadProc(void *arg)
{
    ((Thread *)arg)->Run();
    return NULL;
